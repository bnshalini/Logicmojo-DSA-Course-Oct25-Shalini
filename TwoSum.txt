class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer,Integer> map=new HashMap<>();
        int[] arr=new int[2];
        for(int i=0;i<nums.length;i++){
            int ans=target-nums[i];
            if(map.containsKey(ans)){
                arr[1]=map.get(ans);
                arr[0]=ans;
                return arr;
            }
            map.put(ans);
        }
        
    }
}
